  <script>
    function responder(respuesta) {
      const mensaje = document.getElementById("respuesta");
      let texto = "";

      if (respuesta === "si") {
        texto = "💌 ¡Yay! ¡Daylin dijo que SÍ! Te amo ❤️";
      } else if (respuesta === "no") {
        texto = "😢 Ay cómo lamento eso... Yo en verdad quiero contigo 💔";
      } else {
        texto = "💗 No importa mi Day... yo te esperaría una eternidad, porque sé que tú eres la indicada.";
      }

      mensaje.innerHTML = texto;
    }

    const heartsContainer = document.querySelector(".hearts");
    for (let i = 0; i < 30; i++) {
      const heart = document.createElement("div");
      heart.classList.add("heart");
      heart.style.left = Math.random() * 100 + "vw";
      heart.style.animationDuration = 3 + Math.random() * 5 + "s";
      heart.style.top = Math.random() * -100 + "px";
      heartsContainer.appendChild(heart);
    }
